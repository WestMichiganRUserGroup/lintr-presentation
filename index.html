system.time(expr2 = rnorm(10e5))
						

system.time(expr1 <- rnorm(10e5))
The first is an error because = is argument passing in that context! 

x[x<-1]
						

x[x < -1]
x is 1. Whoops! We want all x es less than -1. 

x <- 1 : 10
base :: get
						

x <- 1:10
base::get
x is 1. Whoops! We want all x es less than -1.

{
y}

  x}

  x else
    y
						

if (x < 5){
  x

  y

if (is.null(ylim)) ylim <- c(0, 0.06)
